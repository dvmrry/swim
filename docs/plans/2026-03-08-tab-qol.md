### Task 1: `gt` — Jump to Tab by Number

Add `gt` keybinding that uses the count prefix to jump to tab N. `3gt` = tab 3, `gt` alone = next tab. If N exceeds tab count, jump to last tab.

**Files:**

**Step 1: Add keybinding**

In `input.c`, in `mode_init`, add after the `"."` binding (line 108):

```c

```

**Step 2: Add action handler**

In `main.m`, in `handle_action`, add after the `"next-tab"` block (around line 166):

```c
    } else if (strcmp(action, "goto-tab") == 0) {
        int target;
        if (app.mode.count > 0) {
            target = app.mode.count - 1;  // 1-indexed to 0-indexed
            if (target >= app.browser.tab_count) target = app.browser.tab_count - 1;
            if (target < 0) target = 0;
        } else {
            // No count = next tab (same as K)
            target = app.browser.active_tab + 1;
            if (target >= app.browser.tab_count) target = 0;
        }
        browser_set_active(&app.browser, target);
        ui_select_tab(app.ui, target);
        Tab *t = browser_active(&app.browser);
        if (t && t->lazy && t->url[0]) { t->lazy = false; ui_navigate(app.ui, t->url); }
        sync_tab_display();
```

Note: The count prefix is read from `app.mode.count` which is set by the key handler before the action fires, and cleared after. `handle_action` at the top has `int count = get_count();` but `goto-tab` reads `app.mode.count` directly because it needs 0 (no count) vs 1+ (explicit count) distinction.

```c
void browser_move_tab(Browser *b, int from, int to) {
    if (from < 0 || from >= b->tab_count) return;
    if (to < 0 || to >= b->tab_count) return;
    if (from == to) return;

    Tab tmp = b->tabs[from];
    if (from < to) {
        for (int i = from; i < to; i++) b->tabs[i] = b->tabs[i + 1];
    } else {
        for (int i = from; i > to; i--) b->tabs[i] = b->tabs[i - 1];
    }
    b->tabs[to] = tmp;

    // Update active_tab to follow the moved tab
    if (b->active_tab == from) {
        b->active_tab = to;
    }
}
```

```objc
static void cmd_tabs(const char *args, void *ctx) {
    (void)args; (void)ctx;
    if (app.browser.tab_count == 0) return;

    char buf[2048] = {0};
    int pos = 0;
    for (int i = 0; i < app.browser.tab_count && pos < (int)sizeof(buf) - 100; i++) {
        Tab *t = &app.browser.tabs[i];
        const char *title = t->title[0] ? t->title : t->url;
        // Truncate title
        char short_title[30];
        snprintf(short_title, sizeof(short_title), "%s", title);

        if (i > 0) pos += snprintf(buf + pos, sizeof(buf) - pos, " | ");
        if (i == app.browser.active_tab) {
            pos += snprintf(buf + pos, sizeof(buf) - pos, "%d: *%s", i + 1, short_title);
        } else {
            pos += snprintf(buf + pos, sizeof(buf) - pos, "%d: %s", i + 1, short_title);
        }
    }
    ui_set_status_message(app.ui, buf);
}
```

**Step 2: Add `:tabclose` command**

```objc
static void cmd_tabclose(const char *args, void *ctx) {
    (void)ctx;
    int target;
    if (args && args[0]) {
        target = atoi(args) - 1;  // 1-indexed to 0-indexed
    } else {
        target = app.browser.active_tab;  // no arg = current tab
    }

    if (target < 0 || target >= app.browser.tab_count) {
        ui_set_status_message(app.ui, "Invalid tab number");
        return;
    }

    if (app.browser.tab_count <= 1) {
        ui_close(app.ui);
        return;
    }

    browser_close_tab(&app.browser, target);
    ui_close_tab(app.ui, target);
    browser_set_active(&app.browser, app.browser.active_tab);
    sync_tab_display();
}
```

**Step 3: Add `:tabonly` command**

```objc
static void cmd_tabonly(const char *args, void *ctx) {
    (void)args; (void)ctx;
    int keep = app.browser.active_tab;

    // Close tabs from end to start, skipping the active one
    for (int i = app.browser.tab_count - 1; i >= 0; i--) {
        if (i == keep) continue;
        browser_close_tab(&app.browser, i);
        ui_close_tab(app.ui, i);
        // Adjust keep index if we removed a tab before it
        if (i < keep) keep--;
    }
    browser_set_active(&app.browser, 0);  // only one tab left, index 0
    sync_tab_display();
}
```
